(function() {
  'use strict';
 
  angular
    .module('romanescuAdmin')
    .config(routerConfig);
 
  /** @ngInject */
  function routerConfig($stateProvider, $urlRouterProvider) {
    $stateProvider
 
      .state('index', {
        abstract: true,
        url: "/index",
        templateUrl: "app/components/common/content.html"
      })
      .state('index.main', {
        url: "/main",
        templateUrl: "app/components/main/main.html",
        data: { pageTitle: 'Dashboard' }
      })
      .state('index.products', {
        url: "/products",
        templateUrl: "app/components/products/list.html",
        data: { pageTitle: 'Products' }
      })
 
 
      .state('index.companies', {
          url: "/companies",
          templateUrl: "app/components/companies/list.html",
          data: { pageTitle: 'Companies' }
      })
      .state('index.companies_new', {
          url: "/companies_new",
          templateUrl: "app/components/companies/new.html",
          data: { pageTitle: 'New Companie' }
      })
      .state('index.companies_dashboard', {
          url: "/companies_dashboard",
          templateUrl: "app/components/companies/dashboard.html",
          data: { pageTitle: '$CompanieName' }
      })
 
 
      .state('index.users', {
        url: "/users",
        templateUrl: "app/components/users/list.html",
        data: { pageTitle: 'Users' }
      });
 
    $urlRouterProvider.otherwise('/index/main');
  }
 
})();
